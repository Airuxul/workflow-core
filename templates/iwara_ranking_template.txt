<script>
const viewData = {view_data};

function renderTable(data) {{
    let html = '<table><tr><th>排名</th><th>标题</th><th>播放量</th><th>点赞</th><th>视频链接</th><th>封面</th></tr>';

        html += `<tr>`+
            `<td>{{i+1}}</td>`+
            `<td>{{v.title||''}}</td>`+
            `<td>{{v.views||''}}</td>`+
            `<td>{{v.likes||''}}</td>`+
            `<td><a href=\"https://www.iwara.tv/video/{{v.video_id}}\" target=\"_blank\">链接</a></td>`+
            `<td><img src=\"{{v.img_url||''}}\"></td>`+
        `</tr>`;
    }});
    html += '</table>';
    return html;
}}
function showTable(type) {{
    document.getElementById('table-container').innerHTML = renderTable(type==='view'?viewData:likesData);
}}
